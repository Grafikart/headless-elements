# Dropdown menu elements `<dropdown-menu>`

This component displays an accessible dropdown menu following the W3 recommandations.

## Usage

After registering the component you have to follow a specific structure

```html
<dropdown-menu>
  <button id="menu">Show more information</button>
  <dropdown-menu-content hidden>
    <!-- Your list of link or any arbitrary HTML -->
  </dropdown-menu-content>
</dropdown-menu>
```

## Demo

export const Template = ({label, ...args}) => {
  try {
    DropdownMenu.register();
  } catch (e) {
    // do nothing this is custom elements
  }
  const div = document.createElement('div')
  div.style.setProperty('height', '230px')
  div.innerHTML = `
   <dropdown-menu>
    <button id="menu">Show more information</button>
      <dropdown-menu-content hidden>
        <ul>
          <li><a id="first-link" href="#">First link</a></li>
          <li><a id="second-link" href="#">Second link</a></li>
        </ul>
        <hr>
        <ul>
          <li><a id="last-link" href="#">Third link</a></li>
        </ul>
     </dropdown-menu-content>
    </dropdown-menu>
  `
  return div
};
